% ---- RIGHT: Methodology ----
\begin{column}{\colwidth}

  \begingroup
  \setbeamercolor{block title}{fg=white, bg=blockpurple}
  \setbeamercolor{block body}{fg=darktext, bg=white}
  \begin{block}{Methodology: Temporal Vendi Score}
    \LARGE
    \begin{enumerate}
      \item \textbf{Sample Trajectories}  via environment rollouts.

      \vspace{0.3cm}
      \item \textbf{Measure Similarity} between $\tau{=}(s_1,\!\dots,\!s_L)$ and $\tau'{=}(s'_1,\!\dots,\!s'_{L'})$ via the \textbf{Global Alignment Kernel \cite{cuturi2007kernel}}:

      \vspace{0.4cm}
      \begin{center}
        \scalebox{1.35}{$\displaystyle
          k(\tau,\tau') \;=\!\!\sum_{\pi \in \mathcal{A}(\tau,\tau')} \prod_{(i,j)\in\pi} \exp\!\left(-\frac{d_{\text{sym}}(s_i, s'_j)}{\sigma}\right)
        $}
      \end{center}
      \vspace{0.3cm}

      $\mathcal{A}(\tau,\tau')$ = all valid monotonic alignments; $d$ is a symmetrized \textbf{time-to-reach} distance between states.

      \vspace{0.3cm}
      \item \textbf{Aggregate:} Eigenvalues of the pairwise similarity matrix     
      \item \textbf{Score:} interpretable diversity scalar via the \textbf{Vendi Score} \cite{friedman2023vendiscorediversityevaluation}.
    \end{enumerate}
  \end{block}
  \endgroup

\end{column}

    \begin{block}{About the Experiments}
      \footnotesize
		\LARGE
      \textbf{\textcolor{blockheader}{Interpretability.}}
      In a grid maze \metric{} returns 1.0 for a single path, 1.67 for four overlapping paths on one side, 1.99 for two paths on opposite sides, and 3.32 for the mirrored configuration (${\approx}2\times1.67$).

      \vspace{0.4em}

      \textbf{\textcolor{blockheader}{Superiority Over Baselines.}}
      As the agent learns to use 1–5 keys in a maze, state coverage and occupancy entropy stay nearly flat. The diversity is temporal, not spatial. \metric{} correctly captures this sequential structure and increases steadily with each new strategy.

      \vspace{0.4em}

      \textbf{\textcolor{blockheader}{Continuous Control.}}
      \metric{} scales to continuous point-maze environments using learned quasimetric distances. It stabilises within $\sim$256 rollouts and cleanly separates low and high-diversity.
    \end{block}
