\begin{block}{Problem \& Context}

\textbf{Motivation.} Autonomous ships must navigate complex coastal environments while avoiding land and other vessels in compliance with the International Regulations for Preventing Collisions at Sea (COLREGs). Safe, rule-compliant decision-making is essential
for real-world deployment of Maritime Autonomous Surface Ships.

\textbf{Goal.} Re-implement in Python a real-world autonomous ship navigation
framework originally developed in MATLAB and validated in a peer-reviewed journal (Journal of
Marine Science and Engineering, 2025).

\begin{block}{Model of the system}


\textbf{State.} The own ship state is
$s_t = (x_t, y_t, \psi_t, v_t)$, where $(x,y)$ is the planar position,
$\psi$ the heading, and $v$ the speed.

\textbf{Control.} The control input is the heading change
$u_t = \Delta \psi_t$, bounded by a maximum turning rate.

\textbf{Dynamics.} A discrete-time kinematic ship model is used:
\[
x_{t+1} = x_t + v_t \cos(\psi_t)\Delta t,\quad
y_{t+1} = y_t + v_t \sin(\psi_t)\Delta t,\quad
\psi_{t+1} = \psi_t + \Delta \psi_t.
\]

\textbf{Environment.} Other vessels are modeled as dynamic obstacles
with constant speed and heading over the prediction horizon, yielding straight-line trajectories.
\[
\begin{aligned}
x^{(j)}_{t+1} &= x^{(j)}_t + v^{(j)}\cos(\psi^{(j)})\,\Delta t, \quad
y^{(j)}_{t+1} &= y^{(j)}_t + v^{(j)}\sin(\psi^{(j)})\,\Delta t.
\end{aligned}
\]

\textbf{Objectives.} Track the reference waypoints while minimizing
control effort and maintaining safe separation from other vessels.

\textbf{Safety constraints.} A collision zone and a COLREG awareness zone
define when collision avoidance and rule-compliant maneuvers are required.

\begin{alertblock}{Control / MPC}
\begin{itemize}
    \item \textbf{Global planning:} a grid-based Theta* planner computes a collision-free
      sequence of waypoints $\{w_i\}$ from map data.
  \item \textbf{Inputs:} current state $s_t$, next waypoint $w_{i+1}$, and nearby vessel states $\{s_t^{(j)}\}$.
    \item \textbf{Output:} a horizon-$H$ sequence of heading changes $\{\Delta\psi_{t+h}\}_{h=0}^{H-1}$ ensuring collision-free behavior.

\end{itemize}

\end{alertblock}

\end{column}

\begin{column}{1.02\colwidth}
\begin{alertblock}{Control / MPC}
\begin{itemize}
\item \textbf{MPC.}
At each time step, a finite set of admissible heading changes is defined:
\[
\mathcal{U} = \{\Delta\psi^{(1)}, \ldots, \Delta\psi^{(M)}\}
\subset [-\Delta\psi_{\max}, \Delta\psi_{\max}].
\]

Each $\Delta\psi \in \mathcal{U}$ is applied as a constant turning rate over
a prediction horizon $H$, inducing a candidate trajectory
\[
\tau(\Delta\psi) = \{s_{t+h}\}_{h=1}^{H}, \qquad
s_{t+h+1} = f\!\left(s_{t+h}, \Delta\psi\right).
\]

Candidate trajectories not satisfying the minimum-distance constraint
\[
\|s_{t+h} - s^{(j)}_{t+h}\| \ge d_{\text{collision}},
\quad \forall h,\ \forall j,
\]
are discarded, yielding a feasible set
$\mathcal{T}_{\mathrm{feasible}} \subseteq\{\tau(\Delta\psi)\mid\Delta\psi\in\mathcal{U}\}$.

Among $\mathcal{T}_{\mathrm{feasible}}$, the trajectory minimizing deviation
from the next waypoint direction is selected.
  \item \textbf{Receding horizon:} apply only the first control $\Delta\psi_t$, then re-solve at $t\!+\!1$ with updated observations.
\end{itemize}
\end{alertblock}

\begin{block}{Limitations}
\begin{itemize}
    \item \textbf{Simplified environment:} the simulation does not account for
    complex factors such as weather conditions, partial observability,
    or vessel size and maneuverability differences.
  \item \textbf{Limited control authority:} the controller acts only on steering
  (heading changes). In dense traffic scenarios (e.g., port exits),
  speed control would be necessary to ensure safety.
  \item \textbf{Discretized control:} the simplified MPC evaluates a finite set of
  constant turning rates, which trades optimality for real-time feasibility.
\end{itemize}
\end{block}

\begin{block}{What We Learned}
\begin{itemize}
    \item \textbf{Key insight:} combining global planning with predictive,
  constraint-aware local control is sufficient to obtain safe and COLREG-compliant
  behavior in complex scenarios.
  \item \textbf{Geographic data handling:} Working with real geographic data requires careful handling of coordinate
  systems, local projections, hemispheres, and coastline representations based
  on polygons.
  \item \textbf{Visualization engineering:} Building clear visualizations and animations (Matplotlib) required careful engineering for debugging and validation.
\end{itemize}
\end{block}
